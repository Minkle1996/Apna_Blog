create database Apna_Blog;
use Apna_Blog;

create table contacts 
(ct_seqno int NOT NULL AUTO_INCREMENT,
ct_name varchar(100),
ct_email varchar(200),
ct_phone varchar(20),
ct_msg varchar(1000),
ct_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
primary key (ct_seqno)

);


create table posts
(
post_seqno int NOT NULL AUTO_INCREMENT,
post_title varchar(1000),
post_content varchar(2000),
post_date  TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
post_slug varchar(100),
post_img_url varchar(500),
post_tagline varchar(500)
primary key (post_seqno)

);
